57    @Test
58    public void smallDictTest() {
59      Trie trie = new Trie();
60      DictReader reader = null;
61      try {
62        reader = new DictReader(new File("./dict/small_sherlock.txt"));
63      } catch (FileNotFoundException e) {
64        System.out.println("ERROR: Small Dict Test");
65      }
66      if (reader != null) {
67        String[] tokens;
68        while ((tokens = reader.parseNextLine("\\W")) !=  null) {
69          for (int i = 0; i < tokens.length; i++) {
70            String t = tokens[i];
71            trie.addWord(t);
72          }
73        }
74      reader.close();
75      }
76      List<String> within = trie.levenshteinSearch("ame", 2);
77      assertTrue(within.size() == 7);
78      assertTrue(within.contains("she"));
79      assertTrue(within.contains("the"));
80      assertTrue(within.contains("have"));
81      assertTrue(within.contains("any"));
82      assertTrue(within.contains("and"));
83      assertTrue(within.contains("the"));
84      assertTrue(within.contains("he"));
85      assertTrue(within.contains("name"));
86      assertFalse(within.contains("irene"));
87      assertFalse(within.contains("holmes"));
88      assertFalse(within.contains("her"));
89      assertFalse(within.contains("love"));
90    }
