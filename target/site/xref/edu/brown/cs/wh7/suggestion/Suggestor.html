1   package edu.brown.cs.wh7.suggestion;

3   import java.util.HashSet;
4   
5   import edu.brown.cs.wh7.Trie.Trie;
6   
7   /**
8    * Class that handles suggestion
9    * generation for user input.
10   * @author wh7
11   *
12   */
13  public final class Suggestor {
14    private Trie trie;
15    private boolean led = false;
16    private boolean prefix = false;
17    private boolean whitespace = false;
18    private int max;
19  
20    /**
21     * Suggestor uses an internal
22     * builder class to manage
23     * the booleans that determine
24     * which types of suggestions
25     * are generated.
26     * @author wh7
27     *
28     */
29    public static class Builder {
30      private Trie trie;
31      private boolean l, p, w;
32      private int mx;
33  
34      /**
35       * Constructor needs access to the
36       * engine and the trie.
37       * @param t the trie
38       */
39      public Builder(Trie t) {
40        this.trie = t;
41      }
42  
43      /**
44       * Setter for LED and maxLed for
45       * suggestion generation.
46       * @param led true to turn on
47       * led generation
48       * @param max edit distance
49       * suggestion
50       * @return Builder to cascade
51       * calls
52       */
53      public Builder setLed(boolean led, int max) {
54        this.l = led;
55        this.mx = max;
56        return this;
57      }
58  
59      /**
60       * Setter for prefix generation.
61       * @param prefix true to
62       * turn on prefix generation
63       * @return Builder to cascade
64       * calls
65       */
66      public Builder setPrefix(boolean prefix) {
67        this.p = prefix;
68        return this;
69      }
70  
71      /**
72       * Setter for whitespace
73       * generation.
74       * @param white true to
75       * turn on whitespace
76       * generation
77       * @return Builder to cascade
78       * calls
79       */
80      public Builder setWhitespace(boolean white) {
81        this.w = white;
82        return this;
83      }
84  
85      /**
86       * Method to create a new
87       * Suggestor.
88       * @return new Suggestor
89       */
90      public Suggestor build() {
91        return new Suggestor(this);
92      }
93    }
94  
95    /**
96     * Private constructor for the
97     * suggestor.
98     * @param b Builder used to
99     * set internal fields
100    */
101   private Suggestor(Builder b) {
102     this.trie = b.trie;
103     this.led = b.l;
104     this.prefix = b.p;
105     this.whitespace = b.w;
106     this.max = b.mx;
107   }
108 
109   /**
110    * A method to factor out the process of
111    * making suggestions based on the trie.
112    * @param toMatch the word to match
113    * @return a set of all the words
114    * generated by the suggestor
115    */
116   public HashSet<String> make(String toMatch) {
117     HashSet<String> suggestions = new HashSet<String>();
118     if (prefix) {
119       suggestions.addAll(trie.matchPrefix(toMatch));
120     }
121     if (whitespace) {
122       suggestions.addAll(trie.matchWhitespace(toMatch));
123     }
124     if (led) {
125       suggestions.addAll(trie.levenshteinSearch(toMatch, max));
126     }
127     return suggestions;
128   }
129 }
