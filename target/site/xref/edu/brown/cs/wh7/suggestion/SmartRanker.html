3   import java.io.Serializable;
4   import java.util.Comparator;
5   
6   /**
7    * This class implements my smart ranking system for the autocorrect program.
8    * @author wh7
9    *
10   * @param <T>
11   *          Type must implement candidate.
12   */
13  public class SmartRanker<T extends Candidate> implements Comparator<T>,
14      Serializable {
15  
16    /**
17     * Compare method to ascertain which candidate has a higher score. Tiebreaker
18     * is alphabetical order.
19     * @param c1
20     *          first candidate to compare
21     * @param c2
22     *          second candidate to compare
23     * @return negative integer if c1 is higher ranked than c2, positive integer
24     *         if c2 is higher ranked than c1.
25     */
26    @Override
27    public int compare(Candidate c1, Candidate c2) {
28      if (c1.getScore() > c2.getScore()) {
29        return -1;
30      }
31      if (c1.getScore() < c2.getScore()) {
32        return 1;
33      }
34      return c1.getPhrase().compareTo(c2.getPhrase());
35    }
36  }
