3   import java.io.BufferedReader;
4   import java.io.File;
5   import java.io.FileNotFoundException;
6   import java.io.FileReader;
7   import java.io.IOException;
8   
9   /**
10   * This class allows a user to read
11   * from a CSV file, line by line.
12   * @author wh7 (Wesley Herts)
13   *
14   */
15  public class CSVReader {
16    private BufferedReader bR;
17    private File csv;
18  
19    /**
20     * Constructor for the CSV Reader.
21     * @param csvFile the file to
22     * read from
23     * @throws FileNotFoundException throws this exception
24     * the file passed to the constructor cannot be found.
25     */
26    public CSVReader(File csvFile) throws FileNotFoundException {
27      this.csv = csvFile;
28      bR = new BufferedReader(new FileReader(csv));
29    }
30    /**
31     * Method to parse the next line
32     * in the CSV file and return it,
33     * delimited by the parameter, delim.
34     * @param delim String to delimit
35     * the next line on.
36     * @return the next line of the CSV
37     * file, tokenized by delim
38     */
39    public String[] parseNextLine(String delim) {
40      // can this parameter go wrong?
41      String line = null;
42      try {
43        line = bR.readLine();
44      } catch (IOException e) {
45        System.out.println("ERROR: I/O Exception found.");
46        this.brCloseWrapper();
47      }
48  
49      if (line == null) {
50        return null;
51      }
52      String[] tokens = line.split(delim);
53      return tokens;
54    }
55  
56    /**
57     * Method to close the file reader,
58     * simply calls the private method
59     * that wraps up closing the internal
60     * BufferedReader.
61     */
62    public void close() {
63      brCloseWrapper();
64    }
65    /**
66     * A private method to factor out
67     * the try-catch statements that
68     * are necessary to reading and
69     * closing the BufferedReader.
70     */
71    private void brCloseWrapper() {
72      try {
73        bR.close();
74      } catch (IOException e1) {
75        System.out.println("ERROR: I/O Exception caught closing Buffered Reader");
76        e1.printStackTrace();
77      }
78    }
79  }
