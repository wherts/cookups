79    /**
80     * Private method to execute
81     * the program.
82     */
83    private void run() throws Exception {
84      OptionParser parser = new OptionParser();
85      //setting up parser acceptances
86      parser.accepts("led").withRequiredArg().ofType(Integer.class);
87      parser.accepts("prefix");
88      parser.accepts("whitespace");
89      parser.accepts("smart");
90      parser.accepts("gui");
91  
92      OptionSpec<File> fileSpec = parser.nonOptions().ofType(File.class);
93      OptionSet options = null;
94      try {
95        options = parser.parse(args);
96      } catch (Exception e) {
97        throw new Exception("ERROR: Given unrecognized option");
98      }
99  
100     //OptionSet has all the
101     List<File> corpus = options.valuesOf(fileSpec);
102     if (corpus.size() == 0) {
103       throw new Exception("ERROR: Please specify a corpus file");
104     }
105     eng = new Engine();
106     myTrie = new Trie();
107 
108     maxLed = 0;
109     boolean hasFile = false;
110 
111     for (File f : corpus) {
112       hasFile = true;
113       try {
114         readFile(f);
115       } catch (Exception e) {
116         throw new Exception(e.getMessage());
117       }
118     }
119 
120     led = false;
121     prefix = false;
122     whitespace = false;
123     smartRank = false;
124     if (options.has("led")) {
125       maxLed = (int) options.valueOf("led");
126       if (maxLed <= 0) {
127         throw new Exception("ERROR: LED distance must be greater than 0");
128       }
129       led = true;
130     }
131     if (options.has("prefix")) {
132       prefix = true;
133     }
134     if (options.has("whitespace")) {
135       whitespace = true;
136     }
137     if (options.has("smart")) {
138       smartRank = true;
139     }
140 
141     suggestor = new Suggestor.Builder(myTrie)
142                               .setLed(led, maxLed)
143                               .setWhitespace(whitespace)
144                               .setPrefix(prefix)
145                               .build();
146 
147     if (options.has("gui")) {
148       runSparkServer(suggestor, myTrie, eng);
149     } else if (hasFile) {
150       BufferedReader bR =
151           new BufferedReader(new InputStreamReader(System.in));
152       String nextLine = null;
153       System.out.println("Ready");
154       while ((nextLine = brReadWrapper(bR)) != null) {
155         if (!nextLine.endsWith(" ")) {
156           //ignore line ending in space
157           if (nextLine.length() == 0) {
158             break;
159           } else {
160             String[] tokens = nextLine.split("\\W");
161             if (tokens.length > 0) {
162               List<String> fixed = fixTokens(tokens);
163               String prev = "";
164               String toMatch = "";
165               if (fixed.size() > 1) {
166                 prev = fixed.get(fixed.size() - 2);
167                 toMatch = fixed.get(fixed.size() - 1);
168               } else if (fixed.size() == 1) {
169                 toMatch = fixed.get(0);
170               }
171               HashSet<String> suggestions = suggestor.make(toMatch);
172               rankAndPrint(suggestions, fixed, prev, toMatch);
173             } //ends if
174           } //ends else
175         }
176         System.out.println("");
177       } //end while loop
178       brCloseWrapper(bR);
179     } else {
180       throw new Exception("ERROR: No file specified");
181     }
182   }

184   /**
185    * Method to rank and print suggestions
186    * to System.out if the user has not
187    * activated the GUI.
188    * @param toMatch word to find matches
189    * to
190    * @param prev the word preceding
191    * toMatch
192    * @param fixed list of strings
193    * from the CLI that has been
194    * cleaned of non-alphabetic characters
195    * @param suggestions generated by
196    * the suggestor
197    */
198   private void rankAndPrint(HashSet<String> suggestions,
199       List<String> fixed,
200       String prev,
201       String toMatch) {
202     int toPrint = Constants.PRINT_NUM;
203     if (myTrie.hasWord(toMatch)) {
204       for (int k = 0; k < fixed.size() - 1; k++) {
205         String str = fixed.get(k);
206         System.out.printf("%s ", str);
207       }
208       System.out.printf("%s%n", toMatch);
209       suggestions.remove(toMatch);
210       toPrint = Constants.PRINT_NUM - 1;
211     }
212 
213     List<Candidate> candidates = new ArrayList<Candidate>();
214     if (!smartRank) {
215       candidates = eng.makeBasicCandidates(suggestions, prev);
216       Collections.sort(candidates, new Ranker<Candidate>());
217     } else {
218       candidates = eng.makeSmartCandidates(suggestions, prev, toMatch);
219       Collections.sort(candidates, new SmartRanker<Candidate>());
220     }
221     int len = Math.min(toPrint, candidates.size());
222     for (int i = 0; i < len; i++) {
223       for (int j = 0; j < fixed.size() - 1; j++) {
224         String str = fixed.get(j);
225         System.out.printf("%s ", str);
226       }
227       System.out.printf("%s%n", candidates.get(i).getPhrase());
228     }
229   }

244   /**
245    * A helper method to clean up user input.
246    * @param the tokenized input string
247    * @return a list cleaned up strings from
248    * input
249    */
250   private static List<String> fixTokens(String[] tokens) {
251     List<String> fixed = new ArrayList<String>();
252     for (String s : tokens) {
253       s = s.replaceAll("[^a-zA-Z ]", "");
254       if (s.length() > 0) {
255         s = s.toLowerCase();
256         fixed.add(s);
257       }
258     }
259     return fixed;
260   }

262   /**
263    * A wrapper to factor out exception catching when reading through
264    * BufferedReader.
265    * @return A string containing the next line from the BufferedReader
266    */
267   private static String brReadWrapper(BufferedReader bR) {
268     String line = null;
269     try {
270       line = bR.readLine();
271     } catch (IOException e) {
272       System.err.println(e.getMessage());
273       try {
274         bR.close();
275       } catch (IOException e1) {
276         System.err.println(e1.getMessage());
277       }
278     }
279     return line;
280   }

282   private void readFile(File file) throws Exception {
283     DictReader reader = null;
284     try {
285       reader = new DictReader(file);
286     } catch (FileNotFoundException e) {
287       throw new Exception("ERROR: Input File Not Found");
288     }
289     if (reader != null) {
290       String[] tokens;
291       while ((tokens = reader.parseNextLine("\\W+")) !=  null) {
292         if (tokens.length > 0) {
293           String prev = tokens[0];
294           prev = prev.toLowerCase();
295           eng.addUnigram(prev);
296           myTrie.addWord(prev);
297           for (int i = 1; i < tokens.length; i++) {
298             String t = tokens[i];
299             if (t.length() > 0) {
300               String curr = t.toLowerCase();
301               myTrie.addWord(curr);
302               eng.addUnigram(curr);
303               eng.addBigram(prev, curr);
304               prev = curr;
305             }
306           }
307         }
308       }
309       reader.close();
310     }
311   }

314    * Private method to activate the GUI.
315    * @param s the suggestor for this
316    * instance of autocorrect
317    * @param t the trie for this instance
318    * of autocorrect
319    * @param e the engine for this instance
320    * of autocorrect

322   private static void runSparkServer(Suggestor s, Trie t, Engine e) {
323     // We need to serve some simple static files containing CSS and JavaScript.
324     // This tells Spark where to look for urls of the form "/static/*".
325     Spark.setPort(Constants.PORT);
326     Spark.externalStaticFileLocation("src/main/resources/static");
327     Spark.get("/autocorrect", new StartHandler(), new FreeMarkerEngine());
328     Spark.post("/suggestions", new SuggestionHandler(s, t, e));
329   }

351   /**
352    * Class to handle generating suggestions based
353    * on user input.
354    * @author wh7
355    *
356    */
357   private static final class SuggestionHandler implements Route {
358     private Suggestor suggestor;
359     private Trie myTrie;
360     private Engine eng;
361 
362     /**
363      * Constructor for class handling post requests.
364      * @param s the suggestor for this instance of
365      * autocorrect
366      * @param t the trie for this instance of
367      * autocorrect
368      * @param e the engine for this instance of
369      * autocorrect
370      */
371     public SuggestionHandler(Suggestor s, Trie t, Engine e) {
372       this.suggestor = s;
373       this.myTrie = t;
374       this.eng = e;
375     }
376 
377     /**
378      * Method for handling post requests.
379      * @return an object that contains
380      * all the suggested strings
381      */
382     @Override
383     public Object handle(final Request req, final Response res) {
384       QueryParamsMap qm = req.queryMap();
385       String str = qm.value("input");
386       String[] tokens = str.split("\\W");
387 
388       if (tokens.length > 0) {
389         List<String> fixed = fixTokens(tokens);
390         String prev = "";
391         String toMatch = "";
392         if (fixed.size() > 1) {
393           prev = fixed.get(fixed.size() - 2);
394           toMatch = fixed.get(fixed.size() - 1);
395         } else if (fixed.size() > 0) {
396           toMatch = fixed.get(0);
397         }
398         HashSet<String> suggestions = suggestor.make(toMatch);
399         List<String> ranked = new ArrayList<String>();
400         if (suggestions.size() > 0) {
401           ranked = rank(suggestions, fixed, prev, toMatch);
402         }
403         Map<String, Object> variables
404           = ImmutableMap.of("ranked", ranked);
405         return GSON.toJson(variables);
406       } //ends if
407       return GSON.toJson(new ArrayList<String>());
408     }
409 
410     /**
411      * Method to rank the suggestions
412      * before posting them.
413      * @param toMatch word to find matches
414      * to
415      * @param prev the word preceding
416      * toMatch
417      * @param fixed list of strings
418      * from the CLI that has been
419      * cleaned of non-alphabetic characters
420      * @param suggestions generated by
421      * the suggestor
422      */
423     private List<String> rank(HashSet<String> suggestions,
424         List<String> fixed,
425         String prev,
426         String toMatch) {
427       List<String> toReturn = new ArrayList<String>();
428       int toPrint = Constants.PRINT_NUM;
429       StringBuilder sB = new StringBuilder();
430       if (myTrie.hasWord(toMatch)) {
431         for (int k = 0; k < fixed.size() - 1; k++) {
432           sB.append(fixed.get(k));
433           sB.append(" ");
434         }
435         sB.append(toMatch);
436         suggestions.remove(toMatch);
437         toReturn.add(sB.toString());
438         toPrint = Constants.PRINT_NUM - 1;
439       }
440 
441       List<Candidate> candidates = new ArrayList<Candidate>();
442       if (!smartRank) {
443         candidates = eng.makeBasicCandidates(suggestions, prev);
444         Collections.sort(candidates, new Ranker<Candidate>());
445       } else {
446         candidates = eng.makeSmartCandidates(suggestions, prev, toMatch);
447         Collections.sort(candidates, new SmartRanker<Candidate>());
448       }
449       int len = Math.min(toPrint, candidates.size());
450       for (int i = 0; i < len; i++) {
451         sB = new StringBuilder();
452         for (int j = 0; j < fixed.size() - 1; j++) {
453           sB.append(fixed.get(j));
454           sB.append(" ");
455         }
456         sB.append(candidates.get(i).getPhrase());
457         toReturn.add(sB.toString());
458       }
459       return toReturn;
460     }
